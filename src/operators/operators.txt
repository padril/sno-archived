representation (arg)
    override string:
        std::wcout << L'"' + arg + L'"' + L'\n';
        return arg;
    override set:
        std::visit([](auto x) {std::wcout << x << '\n'; }, arg);
        return arg;
    default:
        std::wcout << arg << '\n';
        return arg;

print (arg)
    override null:
        std::wcout << '\n';
        return arg;
    override set:
        std::visit([](auto x) {std::wcout << x << '\n'; }, arg);
        return arg;
    default:
        std::wcout << arg << '\n';
        return arg;

debug_print (arg)
    string:
        int t = 0;
        Phrase p = Expression::parse(arg, &t);
        std::wcout << p.tree();
        return arg;

plus (x)
    numeric: return +x;
plus (x, y)
    promote numeric numeric: return x + y;
    string string: return x + y;

minus (x)
    numeric: return -x;
minus (x, y)
    promote numeric numeric: return x - y;

times (x, y)
    promote numeric numeric: return x * y;

slash (x, y)
    override int int: return Rational{x, y};
    promote numeric numeric: return x / y;
